-- EPIC 11: Modulo Jefe de Proyectos (PMO)
-- Fecha: 2026-01-30
-- Descripcion: Tablas base para gestion de proyectos, costos y bitacora IA.

-- 1. Tabla de Proyectos (Cabecera)
CREATE TABLE IF NOT EXISTS pmo_proyectos (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(200) NOT NULL,
    cliente_id INT, -- Referencia opcional a tabla de clientes externa o texto
    cliente_nombre VARCHAR(200), -- Backup texto si no hay ID
    estado VARCHAR(50) DEFAULT 'borrador', -- borrador, ejecucion, bloqueado, cerrado
    presupuesto_venta DECIMAL(12,2) DEFAULT 0,
    fecha_inicio DATE,
    fecha_fin_estimada DATE,
    jefe_proyecto_id INT, -- Usuario responsable
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 2. Tabla de Costos (Items de gasto)
CREATE TABLE IF NOT EXISTS pmo_costos (
    id SERIAL PRIMARY KEY,
    proyecto_id INT REFERENCES pmo_proyectos(id) ON DELETE CASCADE,
    tipo_item VARCHAR(50), -- personal, vehiculo, material, colacion, tag, otros
    descripcion VARCHAR(200),
    monto_estimado DECIMAL(10,2) DEFAULT 0,
    monto_real DECIMAL(10,2) DEFAULT 0,
    estado_aprobacion VARCHAR(20) DEFAULT 'pendiente', -- pendiente, aprobado, rechazado
    created_at TIMESTAMP DEFAULT NOW()
);

-- 3. Tabla de Bitacora IA (Router de informacion)
CREATE TABLE IF NOT EXISTS pmo_bitacora_ia (
    id SERIAL PRIMARY KEY,
    proyecto_id INT REFERENCES pmo_proyectos(id) ON DELETE SET NULL,
    origen VARCHAR(20), -- mail, meet, manual, whatsapp
    contenido_raw TEXT, -- Texto original (resumen o body del mail)
    resumen_ia TEXT, -- Lo que la IA entendio
    acciones_json JSONB, -- Estructura de acciones propuestas: { "bodega": [...], "erp": [...] }
    estado_procesamiento VARCHAR(20) DEFAULT 'pendiente', -- pendiente, procesado, descartado
    created_at TIMESTAMP DEFAULT NOW()
);

-- Indices basicos
CREATE INDEX IF NOT EXISTS idx_pmo_proyectos_estado ON pmo_proyectos(estado);
CREATE INDEX IF NOT EXISTS idx_pmo_costos_proyecto ON pmo_costos(proyecto_id);
CREATE INDEX IF NOT EXISTS idx_pmo_bitacora_proyecto ON pmo_bitacora_ia(proyecto_id);
